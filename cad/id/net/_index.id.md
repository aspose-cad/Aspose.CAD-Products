    <h2 class="h2title">
     Konversikan CAD ke PDF
    </h2>
    <p>
     Aspose.CAD untuk .NET menyediakan kemampuan untuk mengonversi gambar CAD yang didukung ke format PDF. Proses konversi sangat dapat disesuaikan, memungkinkan Anda untuk mengonfigurasi aspek rendering, seperti ukuran halaman, warna latar belakang, warna gambar, mode skala, dan sebagainya.
    </p>
    <div class="codeblock" id="code">
     <h3>
      Simpan CAD dalam format PDF - C#
     </h3>
     <pre><code class="cs">// memuat gambar yang akan dikonversi

using (var img = Aspose.CAD.Image.Load(dir + "template.dwg"))

{

    // buat instance CadRasterizationOptions &amp; atur ukuran halaman yang dihasilkan

    var rasterizationOptions = new Aspose.CAD.ImageOptions.CadRasterizationOptions()

    {

        PageSize = new Aspose.CAD.SizeF(1200, 1200),

    };

    // simpan PDF yang dihasilkan

    img.Save(dir + "output.pdf", new Aspose.CAD.ImageOptions.PdfOptions() { VectorRasterizationOptions = rasterizationOptions });

}</code></pre>
    </div>
