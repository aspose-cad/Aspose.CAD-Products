    <h2 class="h2title">
     Converti CAD in PDF
    </h2>
    <p>
     Aspose.CAD per .NET offre la possibilità di convertire i disegni CAD supportati in formato PDF. Il processo di conversione è altamente personalizzabile e ti consente di configurare gli aspetti di rendering, come le dimensioni della pagina, il colore di sfondo, il colore del disegno, la modalità di ridimensionamento e così via.
    </p>
    <div class="codeblock" id="code">
     <h3>
      Salva CAD in formato PDF - C#
     </h3>
     <pre><code class="cs">// caricare il disegno da convertire

using (var img = Aspose.CAD.Image.Load(dir + "template.dwg"))

{

    // crea un'istanza di CadRasterizationOptions &amp; impostare la dimensione della pagina risultante

    var rasterizationOptions = new Aspose.CAD.ImageOptions.CadRasterizationOptions()

    {

        PageSize = new Aspose.CAD.SizeF(1200, 1200),

    };

    // salva il PDF risultante

    img.Save(dir + "output.pdf", new Aspose.CAD.ImageOptions.PdfOptions() { VectorRasterizationOptions = rasterizationOptions });

}</code></pre>
    </div>
