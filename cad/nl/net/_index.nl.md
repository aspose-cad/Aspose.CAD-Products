    <h2 class="h2title">
     Converteer CAD naar PDF
    </h2>
    <p>
     Aspose.CAD voor .NET biedt de mogelijkheid om ondersteunde CAD-tekeningen naar PDF-formaat te converteren. Het conversieproces is in hoge mate aanpasbaar, zodat u de weergave-aspecten kunt configureren, zoals paginaformaat, achtergrondkleur, tekenkleur, schaalmodus, enzovoort.
    </p>
    <div class="codeblock" id="code">
     <h3>
      CAD opslaan in PDF-formaat - C#
     </h3>
     <pre><code class="cs">// laad de te converteren tekening

using (var img = Aspose.CAD.Image.Load(dir + "template.dwg"))

{

    // maak een instantie van CadRasterizationOptions &amp; stel resulterende paginagrootte in

    var rasterizationOptions = new Aspose.CAD.ImageOptions.CadRasterizationOptions()

    {

        PageSize = new Aspose.CAD.SizeF(1200, 1200),

    };

    // resulterende PDF opslaan

    img.Save(dir + "output.pdf", new Aspose.CAD.ImageOptions.PdfOptions() { VectorRasterizationOptions = rasterizationOptions });

}</code></pre>
    </div>
   </div>
