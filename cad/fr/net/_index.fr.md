    <h2 class="h2title">
     Convertir CAO en PDF
    </h2>
    <p>
     Aspose.CAD pour .NET offre la possibilité de convertir les dessins CAO pris en charge au format PDF. Le processus de conversion est hautement personnalisable, vous permettant de configurer les aspects de rendu, tels que la taille de la page, la couleur d'arrière-plan, la couleur de dessin, le mode d'échelle, etc.
    </p>
    <div class="codeblock" id="code">
     <h3>
      Enregistrer la CAO au format PDF - C#
     </h3>
     <pre><code class="cs">// charger le dessin à convertir

using (var img = Aspose.CAD.Image.Load(dir + "template.dwg"))

{

    // créer une instance de CadRasterizationOptions &amp; définir la taille de la page résultante

    var rasterizationOptions = new Aspose.CAD.ImageOptions.CadRasterizationOptions()

    {

        PageSize = new Aspose.CAD.SizeF(1200, 1200),

    };

    // enregistrer le PDF résultant

    img.Save(dir + "output.pdf", new Aspose.CAD.ImageOptions.PdfOptions() { VectorRasterizationOptions = rasterizationOptions });

}</code></pre>
    </div>
   </div>
